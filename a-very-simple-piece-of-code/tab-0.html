var capitals = {
  "CA": {
    "AB": "Edmonton",
    "BC": "Victoria",
    "MB": "Winnipeg",
    // etc.
  },
  "US": {
    "AL": (year) => {
      if (year < 1846) { return "Tuscaloosa"; }
      return "Montgomery"
    },
    "AK": "Juneau",
    "AR": "Phoenix",
    // etc.
  },
}

function capitalOf(country, stateOrProvinceName, year) {
  var m = capitals[country]
  if (m) {
    var capital = m[stateOrProvinceName]
    if (capital) {
      for (; typeof capital == 'function';) {
        capital = capital(year)
      }
      return capital
    }
  }
  return "nil"
}
